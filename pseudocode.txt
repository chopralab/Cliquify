Tree decomposition of molecule (G -> T)

edges = set of bonds which are not in rings
ssr_list = set of rings retrieved from cycle basis

for r1, r2 in ssr_list 
do
    merge r1 and r2 if they share more than 3 rings
end for

for r1 in ssr_list
do
    n = choose appropriate seleNode for ghost edge formation
    add ghost edge to rings using n
end for

clique_list = set of cliques from graph after adding ghost ghost_edges
singleton = atoms being the intersection of three or more clusters from ssr_list UNION edges 
vertices = edges UNION singleton UNION clique_list

return minimum_spanning_tree of (vertices, edges)
